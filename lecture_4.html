set.seed(1)
ex <- rnorm(100, 0, 1)
ey <- rnorm(100, 0, 1)

# random walk (stochastic trend)
Xt <- cumsum(ex)

# deterministic trend
Yt <- 0.1*c(1:100) + ey

Xt <- data.frame(date = c(1:length(Xt)), Xt = Xt)
fig1a <- ggplot(Xt, aes(date)) + 
  geom_line(aes(y = Xt), size = 0.7, color = "seagreen4") +
  labs(title = "Stochastic Trend: Random Walk", 
       x = "t", y = TeX("$X_t$")) +        
  theme(axis.title = element_text(family = "serif"),
        plot.title = element_text(hjust = 0.5, family = "serif"),
        axis.title.y = element_text(angle = 90, vjust = 0.5))

Yt <- data.frame(date = c(1:length(Yt)), Yt = Yt)
fig1b <- ggplot(Yt, aes(date)) + 
  geom_line(aes(y = Yt), size = 0.7, color = "seagreen4") +
  labs(title = "Deterministic Trend: Drift", 
       x = "t", y = TeX("$Y_t$")) +        
  theme(axis.title = element_text(family = "serif"),
        plot.title = element_text(hjust = 0.5, family = "serif"),
        axis.title.y = element_text(angle = 90, vjust = 0.5))

rhox <- acf(Xt$Xt, plot = F, na.action = na.pass)
alpha <- 0.95
conf.lims <- c(-1,1)*qnorm((1 + alpha)/2)/sqrt(rhox$n.used)

fig1c <- rhox$acf %>% 
  as_tibble() %>% mutate(lags = 1:n()) %>% 
  ggplot(aes(x = lags, y = V1)) + scale_x_continuous(breaks = seq(0,41,4)) +
  geom_hline(yintercept = conf.lims, lty = 2, col = "blue") +
  labs(y = "ACF", x = "Lag", title = "Stochastic Trend: Random Walk") +
  geom_segment(aes(xend = lags, yend = 0)) + geom_point(size = 1) +
  theme(axis.title = element_text(family = "serif"),
        plot.title = element_text(hjust = 0.5, family = "serif")) +
  ylim(-0.2,1)

rhoy <- acf(Yt$Yt, plot = F, na.action = na.pass)
alpha <- 0.95
conf.lims <- c(-1,1)*qnorm((1 + alpha)/2)/sqrt(rhoy$n.used)

fig1d <- rhoy$acf %>%

gold <- read_csv("gld.csv") %>%            # read SPDR gold daily data
  select(Date, `Adj Close`) %>%            # keep only "date" and "adj close" 
  rename(X = `Adj Close`) %>%              # rename "adj close" -> "Xd"
  mutate(l.X = lag(X, n = 1L),             # lag X
         R = log(X/l.X),                   # log return
         DX = X - l.X) %>%                 # 1st order difference in X
  filter(is.na(R) == 0) %>%                # remove missing value
  filter(Date >= "2011-01-01") %>%         # use 2011 data 
  filter(Date <= "2011-12-31")

predict.X <- function(t) {
  arima213.X <- arima(gold$X[1:(231 + t)], order = c(2, 1, 3))
  px <- predict(arima213.X, n.ahead = 1)
  c(px$pred, px$pred - 1.96*px$se, px$pred + 1.96*px$se)
}

pred.val <- t(sapply(0:20, predict.X))

real.price <- gold$X[232:252]
date <- gold$Date[232:252]
table <- data.frame(date = date, price = real.price, 
                    pred.price = pred.val[,1], 
                    CI.lb = pred.val[,2], CI.ub = pred.val[,3])
